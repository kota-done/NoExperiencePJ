# クラス設計一覧

## ドメインモデル
### Employee
- id: long
- firstName: String
- lastName: String
- email: String
- status: EmployeeStatus
- clientCompany: ClientCompany

### ClientCompany
- id: long
- companyName: String
- contactEmail: String
- phoneNumber: String
- businessStatus: BusinessStatus

### EmployeeStatus
- ON_ASSIGNMENT
- WAITING
- LEAVE_OF_ABSENCE
- TRAINING

### BusinessStatus
- ACTIVE
- SUSPENDED
- CLOSED

## Service
### EmployeeService
- getAllEmployees(): List<Employee>
- saveEmployee(Employee employee): void
- getEmployeeById(long id): Employee
- deleteEmployeeById(long id): void
- findPaginated(int pageNo, int pageSize, String sortField, String sortDirection): Page<Employee>

### ClientCompanyService
- getAllClientCompanies(): List<ClientCompany>
- saveClientCompany(ClientCompany clientCompany): void
- getClientCompanyById(long id): ClientCompany
- deleteClientCompanyById(long id): void

## Repository
### EmployeeRepository
- findAll(): List<Employee>
- findAll(Pageable pageable): Page<Employee>
- findById(Long id): Optional<Employee>
- save(Employee employee): Employee
- deleteById(Long id): void

### ClientCompanyRepository
- findAll(): List<ClientCompany>
- findById(Long id): Optional<ClientCompany>
- save(ClientCompany clientCompany): ClientCompany
- deleteById(Long id): void

## Controller
### EmployeeController
- viewHomePage(Model model): String
- showNewEmployeeForm(Model model): String
- saveEmployee(Employee employee): String
- showFormForUpdate(long id, Model model): String
- deleteEmployee(long id): String
- findPaginated(int pageNo, String sortField, String sortDir, Model model): String

### ClientCompanyController
- viewClientCompanyList(Model model): String
- showNewClientCompanyForm(Model model): String
- saveClientCompany(ClientCompany clientCompany): String
- showClientCompanyFormForUpdate(long id, Model model): String
- deleteClientCompany(long id): String
